\newcommand{\collectequation}[3]{
    \begin{collect*}{#1}{}{}{\paragraph{#2}}{}
        \begin{equation*}
            \boxed{
                #3
            }
        \end{equation*}
    \end{collect*}
}

\newcommand{\warnbox}[3][0.7\textwidth]{
    \begin{tcolorbox}[
        width=#1, center,
        colback=yellow!15, colframe=yellow!80!orange, coltitle=black,
        title=\textbf{#2}
    ]
        #3
    \end{tcolorbox}
}

\newcommand{\errorbox}[3][0.7\textwidth]{
    \begin{tcolorbox}[
        width=#1, center,
        colback=red!15, colframe=red!80!orange, coltitle=black,
        title=\textbf{#2}
    ]
        #3
    \end{tcolorbox}
}
