\newcommand{\collectequation}[4]{
    \begin{collect*}{#1}{}{}{
        \noindent\textbf{#2} \\ \indent #3
    }{}
        \begin{equation*}
            \boxed{
                #4
            }
        \end{equation*}
    \end{collect*}
}

\newcommand{\warnbox}[3][0.7\textwidth]{
    \begin{tcolorbox}[
        width=#1, center,
        colback=yellow!15, colframe=yellow!80!orange, coltitle=black,
        title=\textbf{#2}
    ]
        #3
    \end{tcolorbox}
}

\newcommand{\errorbox}[3][0.7\textwidth]{
    \begin{tcolorbox}[
        width=#1, center,
        colback=red!15, colframe=red!80!orange, coltitle=black,
        title=\textbf{#2}
    ]
        #3
    \end{tcolorbox}
}

\newcommand{\mathwarn}[1]{
    \tcboxmath[
        colback=yellow!15, colframe=yellow!80!orange,
        top=0pt, bottom=0pt, left=0pt, right=0pt
    ]{
        #1
    }
}
